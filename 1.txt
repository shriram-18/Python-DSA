# a) Linear and Binary Search with User Input

def linear_search(customer_list, target_id):
    for i in range(len(customer_list)):
        if customer_list[i] == target_id:
            return i   # return index if found
    return -1          # return -1 if not found


def binary_search(customer_list, target_id):
    low = 0
    high = len(customer_list) - 1

    while low <= high:
        mid = (low + high) // 2
        guess = customer_list[mid]

        if guess == target_id:
            return mid      # found at index mid
        elif guess < target_id:
            low = mid + 1   # search right half
        else:
            high = mid - 1  # search left half

    return -1               # not found


while True:
    print("\n-------------------- MENU --------------------")
    print("1. Enter Customer List")
    print("2. Linear Search")
    print("3. Binary Search")
    print("4. Exit")

    choice = input("Enter your choice (1-4): ")

    if choice == '1':
        customer_list = list(map(int, input("Enter customer IDs separated by spaces: ").split()))
        print("\nCustomer list entered:", customer_list)

    elif choice == '2':
        if 'customer_list' not in locals():
            print("Please enter the customer list first!")
        else:
            search_id = int(input("\nEnter ID to be searched (Linear Search): "))
            found = linear_search(customer_list, search_id)
            if found != -1:
                print(f"✅ Customer ID {search_id} found in list at index {found}")
            else:
                print(f"❌ Customer ID {search_id} not found in list")

    elif choice == '3':
        if 'customer_list' not in locals():
            print("Please enter the customer list first!")
        else:
            customer_list.sort()
            print("\nSorted customer list for Binary Search:", customer_list)
            search_id = int(input("\nEnter ID to be searched (Binary Search): "))
            found = binary_search(customer_list, search_id)
            if found != -1:
                print(f"✅ Customer ID {search_id} found in list at index {found}")
            else:
                print(f"❌ Customer ID {search_id} does not exist in list")

    elif choice == '4':
        print("Exiting program.")
        break

    else:
        print("Invalid choice. Please enter a number between 1 and 4.")
